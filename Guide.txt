CREATE:

CREATE TABLE USERS (
    ID INT AUTO_INCREMENT PRIMARY KEY,
    USERNAME VARCHAR(50),
    PASSWORD VARCHAR(256)
);

CREATE TABLE GROUP_ITEM (
    ID INT AUTO_INCREMENT PRIMARY KEY,
    GROUP_ID INT,
    GROUP_NAME VARCHAR(50),
    DATE_YEAR INT,
    DATE_MONTH VARCHAR(50),
    FOREIGN KEY (GROUP_ID) REFERENCES USERS(ID) ON DELETE CASCADE
);

CREATE TABLE GROUP_SUB_ITEM (
    ID INT AUTO_INCREMENT PRIMARY KEY,
    ITEM_ID INT,
    SUB_ITEM_NAME VARCHAR(50),
    VALUE INT,
    GIVEN_DATE DATE,
    FOREIGN KEY (ITEM_ID) REFERENCES GROUP_ITEM(ID) ON DELETE CASCADE
);

DELETE:

DELETE FROM USERS;
DELETE FROM GROUP_ITEM;
DELETE FROM GROUP_SUB_ITEM;
